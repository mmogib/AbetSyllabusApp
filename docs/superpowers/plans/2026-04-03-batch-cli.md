# Batch CLI Implementation Plan

Status: Implemented on 2026-04-03. The delivered runtime uses a Vite-built Node CLI with `jsdom`-based XML adapters rather than the earlier `tsx` / `xmldom` sketch shown in some examples below.

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a deterministic local CLI that batch-processes `PDF`, `DOCX`, and `TXT` source files into structured `success/`, `review/`, `report.csv`, and `report.json` outputs without using AI.

**Architecture:** Split browser-specific extraction and DOCX generation into shared pure cores plus thin browser wrappers and Node adapters. Then implement a batch orchestration layer that discovers files, extracts/parses/reviews each one, writes outputs and reports, and exposes a small `npm run batch -- --input ... --output ...` interface.

**Tech Stack:** TypeScript, Node 20+, Vitest, JSZip, pdfjs-dist, xmldom, tsx

```typescript
// src/lib/extract/docxTextCore.ts
import JSZip from 'jszip';
import { normalizeText } from './normalizeText';

export function extractDocxTextFromDocumentXml(
  documentXml: string,
  parser: DOMParser,
): string {
  const document = parser.parseFromString(documentXml, 'application/xml');
  if (document.getElementsByTagName('parsererror').length > 0) {
    throw new Error('Invalid DOCX file: malformed XML');
  }

  const paragraphs = Array.from(document.getElementsByTagName('w:p'))
    .map((paragraphNode) =>
      Array.from(paragraphNode.getElementsByTagName('w:t'))
        .map((node) => node.textContent ?? '')
        .join('')
        .trim(),
    )
    .filter(Boolean);

  return normalizeText(paragraphs.join('\n'));
}

export async function extractDocxTextFromArrayBuffer(
  buffer: ArrayBuffer,
  parser: DOMParser,
): Promise<string> {
  const zip = await JSZip.loadAsync(buffer);
  const documentXml = await zip.file('word/document.xml')?.async('string');

  if (!documentXml) {
    throw new Error('Invalid DOCX file: missing word/document.xml');
  }

  return extractDocxTextFromDocumentXml(documentXml, parser);
}
```

```typescript
// src/lib/extract/pdfTextCore.ts
import { getDocument } from 'pdfjs-dist/legacy/build/pdf.mjs';
import { normalizeText } from './normalizeText';

export async function extractPdfTextFromBytes(
  bytes: Uint8Array,
  options: { disableWorker?: boolean; useWorkerFetch?: boolean } = {},
): Promise<string> {
  const pdf = await getDocument({
    data: bytes,
    disableWorker: options.disableWorker ?? true,
    useWorkerFetch: options.useWorkerFetch ?? false,
  }).promise;

  try {
    const pages: string[] = [];
    for (let pageNumber = 1; pageNumber <= pdf.numPages; pageNumber += 1) {
      const page = await pdf.getPage(pageNumber);
      try {
        const content = await page.getTextContent();
        pages.push(joinPdfTextItems(content.items));
      } finally {
        page.cleanup();
      }
    }

    return normalizeText(pages.join('\n\n'));
  } finally {
    await pdf.destroy();
  }
}
```

```typescript
// src/lib/docx/generateSyllabusDocxCore.ts
import JSZip from 'jszip';
import type { SyllabusDraft } from '../../types/schema';

export async function buildSyllabusDocxBytes(input: {
  draft: SyllabusDraft;
  templateBytes: ArrayBuffer;
  parseXml: (xml: string) => XMLDocument;
  serializeXml: (document: XMLDocument) => string;
}): Promise<Uint8Array> {
  const zip = await JSZip.loadAsync(new Uint8Array(input.templateBytes));
  const documentXml = await zip.file('word/document.xml')?.async('string');
  if (!documentXml) {
    throw new Error('Template DOCX is missing word/document.xml.');
  }

  const document = input.parseXml(documentXml);
  populateTemplateDocument(document, input.draft);
  zip.file('word/document.xml', input.serializeXml(document));

  const coreXml = await zip.file('docProps/core.xml')?.async('string');
  if (coreXml) {
    zip.file('docProps/core.xml', updateCoreTitle(coreXml, input.draft));
  }

  return zip.generateAsync({ type: 'uint8array' });
}
```

```typescript
import type { BatchRecord } from './batchTypes';

function escapeCsvCell(value: string): string {
  if (/[",\n]/.test(value)) {
    return `"${value.replace(/"/g, '""')}"`;
  }

  return value;
}

export function buildCsvReport(records: readonly BatchRecord[]): string {
  const header = [
    'sourceFile',
    'relativeSourcePath',
    'status',
    'termCode',
    'courseNumber',
    'courseTitle',
    'outputFile',
    'unresolvedFieldCount',
    'unresolvedFields',
    'extractedTextFile',
    'errorMessage',
  ];

  const lines = records.map((record) =>
    [
      record.sourceFile,
      record.relativeSourcePath,
      record.status,
      record.termCode,
      record.courseNumber,
      record.courseTitle,
      record.outputFile,
      String(record.unresolvedFieldCount),
      record.unresolvedFields.join('|'),
      record.extractedTextFile,
      record.errorMessage,
    ].map(escapeCsvCell).join(','),
  );

  return [header.join(','), ...lines].join('\n');
}
```

```typescript
// src/cli/nodeAdapters.ts
import { readFile } from 'node:fs/promises';
import { extname } from 'node:path';
import { DOMParser, XMLSerializer } from '@xmldom/xmldom';
import { extractDocxTextFromArrayBuffer } from '../lib/extract/docxTextCore';
import { extractPdfTextFromBytes } from '../lib/extract/pdfTextCore';
import { buildSyllabusDocxBytes } from '../lib/docx/generateSyllabusDocxCore';
import type { SyllabusDraft } from '../types/schema';

export async function extractSourceTextFromPath(filePath: string): Promise<string> {
  const extension = extname(filePath).toLowerCase();

  if (extension === '.txt') {
    return readFile(filePath, 'utf8');
  }

  const fileBytes = await readFile(filePath);
  if (extension === '.docx') {
    const buffer = fileBytes.buffer.slice(fileBytes.byteOffset, fileBytes.byteOffset + fileBytes.byteLength) as ArrayBuffer;
    return extractDocxTextFromArrayBuffer(buffer, new DOMParser() as unknown as DOMParser);
  }

  if (extension === '.pdf') {
    return extractPdfTextFromBytes(new Uint8Array(fileBytes), { disableWorker: true, useWorkerFetch: false });
  }

  throw new Error(`Unsupported file type: ${extension}`);
}

export async function generateDocxBytesForDraft(draft: SyllabusDraft): Promise<Uint8Array> {
  const templateBytes = await readFile(new URL('../../output_template/ABETSyllabusTemplate2.docx', import.meta.url));
  return buildSyllabusDocxBytes({
    draft,
    templateBytes: templateBytes.buffer.slice(templateBytes.byteOffset, templateBytes.byteOffset + templateBytes.byteLength) as ArrayBuffer,
    parseXml: (xml) => new DOMParser().parseFromString(xml, 'application/xml') as unknown as XMLDocument,
    serializeXml: (document) => new XMLSerializer().serializeToString(document as unknown as Node),
  });
}
```

```typescript
// src/cli/batchGenerate.ts
import { resolve } from 'node:path';
import { extractSourceTextFromPath, generateDocxBytesForDraft } from './nodeAdapters';
import { runBatchGenerate } from './batchGenerateCore';

function readFlag(name: string): string | undefined {
  const index = process.argv.indexOf(name);
  return index >= 0 ? process.argv[index + 1] : undefined;
}

async function main(): Promise<void> {
  const inputDir = readFlag('--input');
  const outputDir = readFlag('--output');

  if (!inputDir || !outputDir) {
    throw new Error('Usage: npm run batch -- --input <dir> --output <dir> [--term 252]');
  }

  const result = await runBatchGenerate(
    {
      inputDir: resolve(inputDir),
      outputDir: resolve(outputDir),
      termCode: readFlag('--term'),
      recursive: readFlag('--recursive') !== 'false',
      copyReviewSources: readFlag('--copy-review-sources') !== 'false',
      writeExtractedText: readFlag('--write-extracted-text') !== 'false',
    },
    {
      extractSourceText: extractSourceTextFromPath,
      generateDocxBytes: generateDocxBytesForDraft,
      now: new Date(),
    },
  );

  console.log(
    `Batch complete: total=${result.summary.totalDiscovered} success=${result.summary.success} needs_review=${result.summary.needsReview} failed=${result.summary.failed}`,
  );
}

main().catch((error) => {
  console.error(error instanceof Error ? error.message : String(error));
  process.exitCode = 1;
});
```
